   \begin{flalign*}
   左边 &= ( BC + D + \overline{D}(\overline{B} + \overline{C})(AD + B) )\\
   &= ( BC + D + \overline{D}(\overline{B}AD + \overline{C}AD + \overline{B}B + \overline{C}B) ) （分配律）\\
   &= ( BC + D + \overline{D}(\overline{B}AD + \overline{C}AD + 0 + \overline{C}B) ) （互补律 ( \overline{B}B=0 )）\\
   &= ( BC + D + \overline{D}\cdot0 + \overline{D}\overline{C}B ) （互补律 ( D\overline{D}=0 )）\\
   &= ( BC + D + B\overline{C}\overline{D} )\\
   &= ( B(C + \overline{C}\overline{D}) + D ) （提取公因子 ( B )）\\
   &= ( B(C + \overline{D}) + D ) （吸收律变形 ( A + \overline{A}B = A + B )）\\
   &= ( BC + B\overline{D} + D )\\
   &= ( BC + (B + D) ) （吸收律变形 ( overline{A}B + A = A + B )）\\
   &= ( B(C + 1) + D ) （提取公因子 ( B )）\\
   &= ( B\cdot1 + D ) （( C + 1 = 1 )）\\
   &= ( B + D ) = 右边 （得证）& 
   \end{flalign*}
